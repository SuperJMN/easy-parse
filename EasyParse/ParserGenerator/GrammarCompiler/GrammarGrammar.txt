lexemes:
ignore @'[ \t]+'
ignore @'#[^\n]*'
arrow is @'\->'
lexemes is 'lexemes:'
keywordIgnore is 'ignore'
keywordIs is 'is'
rules is 'rules:'
nonTerminal is '[A-Z][a-zA-Z0-9_]*'
terminal is @'[a-z_][a-zA-Z0-9_]*'
eoln is @'\n([ \t]*(#[^\n]*)?\n)*'
quotedString is '\'((\\[\\\'nrt])|[^\\\'])*\''
verbatimString is '@\'[^\']*\''

rules:
FullGrammar -> Lexemes rules eoln Grammar
FullGrammar -> eoln FullGrammar
Lexemes -> lexemes eoln
Lexemes -> Lexemes LexemePattern eoln
LexemePattern -> terminal keywordIs String
LexemePattern -> keywordIgnore String
Grammar -> Rule eoln
Grammar -> Grammar Rule eoln
Rule -> nonTerminal arrow RuleBody
RuleBody -> Symbol
RuleBody -> RuleBody Symbol
Symbol -> terminal
Symbol -> nonTerminal
String -> quotedString
String -> verbatimString

lexemes:
ignore @'[ \t]+'
ignore @'#[^\n]*'
arrow matches @'\->'
lexemes matches 'lexemes:'
keywordIgnore matches 'ignore'
keywordMatches matches 'matches'
rules matches 'rules:'
nonTerminal matches '[A-Z][a-zA-Z0-9_]*'
terminal matches @'[a-z_][a-zA-Z0-9_]*'
eoln matches @'\n([ \t]*(#[^\n]*)?\n)*'
quotedString matches '\'((\\[\\\'nrt])|[^\\\'])*\''
verbatimString matches '@\'[^\']*\''

rules:
FullGrammar -> Lexemes rules eoln Grammar
FullGrammar -> eoln FullGrammar
Lexemes -> lexemes eoln
Lexemes -> Lexemes LexemePattern eoln
LexemePattern -> terminal keywordMatches String
LexemePattern -> terminal 'matched' String
LexemePattern -> keywordIgnore String
Grammar -> Rule eoln
Grammar -> Grammar Rule eoln
Rule -> nonTerminal arrow RuleBody
RuleBody -> Symbol
RuleBody -> RuleBody Symbol
Symbol -> terminal
Symbol -> nonTerminal
Symbol -> String
String -> quotedString
String -> verbatimString

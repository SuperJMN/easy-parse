lexemes:
ignore @'\s+'
'n' is @'\d+'        # number
'+' is @'[\+\-]'     # additive operator
'*' is @'[\*\/]'     # multiplicative operator
'(' is @'\('         # open parenthesis
')' is @'\)'         # closed parenthesis

rules:
Additive -> Multiplicative
Additive -> Additive + Multiplicative
Additive -> Additive - Multiplicative
Unit -> n
Unit -> ( Additive )
Multiplicative -> Unit
Multiplicative -> Multiplicative * Unit
Multiplicative -> Multiplicative / Unit

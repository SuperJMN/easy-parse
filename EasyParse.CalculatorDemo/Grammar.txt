lexemes:
ignore @'\s+'
number matches @'\d+'
adOp matches @'[\+\-]'
mulOp matches @'[\*\/]'
openParen matches @'\('
closedParen matches @'\)'

rules:
Additive -> Multiplicative
Additive -> Additive adOp Multiplicative
Unit -> number
Unit -> openParen Additive closedParen
Multiplicative -> Unit
Multiplicative -> Multiplicative mulOp Unit
